def distribute_points(G, points):
    nodes= list(G.nodes())
    new_points=[]
    for i in range(len(nodes)):
        new_points.append(0)
    for n in nodes:
        out=list(G.out_edges(n))
        if(len(out)==0):
            new_points=new_points[n]+points[n]
        else:
            share= points[n]/len(out)
            for(src, tgt) in out:
                new_points[tgt]= new_points[tgt]+share
    return new_points
